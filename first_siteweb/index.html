    Exercice : Programme en C

    Voici un exemple de programme en langage C pour remplir une matrice et calculer la somme de ses éléments positifs : 

                
#include <stdio.h>

int main() {
  int matrix[3][3];
  int sum = 0;

   // Remplissage de la matrice
  printf("Entrez les éléments de la matrice (3x3):\n");
  for (int i = 0; i < 3; i++) {
  for (int j = 0; j < 3; j++) {
   printf("Element [%d][%d]: ", i, j);
   scanf("%d", &matrix[i][j]);
        }
    }

  // Calcul de la somme des éléments positifs
  for (int i = 0; i < 3; i++) {
  for (int j = 0; j < 3; j++) {
  if (matrix[i][j] > 0) {
     sum += matrix[i][j];
            }
        }
    }

 // Affichage de la somme
 printf("La somme des éléments positifs de la matrice
              est : %d\n", sum);

    return 0;
}
